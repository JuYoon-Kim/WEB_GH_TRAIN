# 1) 10부터 20 사이의 정수를 모두 더하시오

hap = 0

for i in range(10,21,1):
    hap += i

print(hap)

# 2) 임의의 리스트 안의 숫자를 모두 더하기

hap = 0

list1 = [2,5,1,8,5,3,5,9]

for i in list1:
    hap += i

print(hap)

# 2) 임의의 리스트안에서 짝수만 골라내기

mylist = [2,5,1,8,5,3,5,9]

list1 = []  #홀수
list2 = []  #짝수

for i in mylist:
    if i % 2 == 0:
        list2.append(i)
    else :
        list1.append(i)
    
print(list1)
print(list2)
